SELECT       b.DETNextRDetentionID, a.detentionfacilityclientnumber as facilityclientnumber, a.ClientFirstName, a.ClientMiddleName, a.ClientLastName, a.ClientSuffixName, 
                         a.ClientBirthDate, a.clientracetext, a.clientgendercode, b.DetentionStartDateTime as startdatetime, 
                         b.DetentionEndDateTime as enddatetime, b.DetentionFacilityORIName as FacilityORIName, b.DetentionHoldReasonText as HoldReasonText,
						 b.DetentionEndReasonText as EndReasonText, 
                         c.DetentionOffenseSequenceNumber as OffenseSequenceNumber, c.DetentionOffenseStatuteChapterCode as OffenseStatuteChapterCode,
						 c.DetentionOffenseStatuteSectionCode as OffenseStatuteSectionCode, 
                         c.DetentionOffenseStatuteSequenceCode as OffenseStatuteSequenceCode, c.DetentionOffenseMOCcode as OffenseMOCcode, 
						 c.DetentionOffenseText as OffenseText, 
                         b.DetentionControllingAgencyORIName as ControllingAgencyORIName, e.DetDesc, d.DetentionCourtCaseIdentifier as CourtCaseIdentifier, 
                         d.DetentionCourtCaseSentenceStartDate as CourtCaseSentenceStartDate
FROM            Jail_Detention b INNER JOIN
                         Jail_Client a ON b.detentionfacilityclientnumber = a.detentionfacilityclientnumber INNER JOIN
                         Jail_Offense c ON b.DETNextRDetentionID = c.DETNextRDetentionID LEFT OUTER JOIN
                         Jail_CourtCase d ON c.DETNextRDetentionID = d.DETNextRDetentionID LEFT OUTER JOIN
                         Jail_lkup_dettype e ON b.DetentionTypeCode = e.DetCode
WHERE       clientlastname is not null
